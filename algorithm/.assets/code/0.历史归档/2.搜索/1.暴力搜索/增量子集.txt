#include<cstdio>
#include<algorithm>
int n,ans[20];
void subset(int k){
    int null=1;
    for(int i=0;i<k;i++,null=0)printf((!i)?"%d":" %d",ans[i]);
    if(null)printf("null");
    printf("\n");
    int s=k?ans[k-1]+1:1;
    for(int i=s;i<=n;i++){
        ans[k]=i;
        subset(k+1);
    }
}
int main(){
    scanf("%d",&n);
    subset(0);
    return 0;
}

input:
3
output:

1
1 2
1 2 3
1 3

3
