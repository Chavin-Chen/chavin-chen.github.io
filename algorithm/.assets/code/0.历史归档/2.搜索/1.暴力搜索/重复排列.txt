
#include<cstdio>
#include<algorithm>
int a[20],n,ans[20];
int sea(int k){
    if(k==n){
        for(int i=0;i<n;i++)
            printf((i==n-1)?"%d\n":"%d ",ans[i]);
    }else for(int i=0;i<n;i++)
        if(!i||a[i]!=a[i-1]){ //深搜剪枝1:不重复
            int o=0,t=0;
            for(int j=0;j<n;j++)if(a[j]==a[i])o++;
            for(int j=0;j<k;j++)if(ans[j]==a[i])t++;
            if(t<o){//剪枝2:目标检查
                ans[k]=a[i];
                sea(k+1);
            }
    }
}
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d",a+i);
    std::sort(a,a+n);
    sea(0);
    return 0;
}

3
