int vis[17],ans[16],n,primes[]={2,3,5,7,11,13,17,19,23,31};
int isprime(int x){
    for(int i=0;i<10;i++)if(x==primes[i])return 1;
    return 0;
}
void dfs(int k){
    if(k==n&&isprime(ans[0]+ans[n-1])){
        for(int i=0;i<n;i++)printf(i==n-1?"%d\n":"%d ",ans[i]);
    }else for(int i=2;i<=n;i++){
        if(!vis[i]&&isprime(ans[k-1]+i)){
            vis[i]=1;
            ans[k]=i;
            dfs(k+1);
            vis[i]=0;
        }
    }
}
int main(){
    while(scanf("%d",&n)!=-1){
        ans[0]=1;vis[1]=1;
        dfs(1);
    }
    return 0;
}
