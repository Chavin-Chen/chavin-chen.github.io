#include<cstdio>
#include<cmath>
int euler(int n){
    int m=sqrt(n+0.5),ans=n;		//f(n)=n(1-1/q1)(1-1/q2)(1-1/q3)...(1-1/q4);
    for(int i=2;i<=m;i++)if(n%i==0){	//唯一分解定理
        ans=ans/i*(i-1);
        while(n%i==0)n/=i;
    }
    if(n>1)ans=ans/n*(n-1);//n为素数
    return ans;
}
int main(){
    int n;
    while(scanf("%d",&n)!=-1)
        printf("%d\n",euler(n));
    return 0;
}
/*
求1~n中与n互素数的个数(欧拉函数值)
input:
2700
output:
720
