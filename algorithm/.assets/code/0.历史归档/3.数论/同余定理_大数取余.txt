#include<cstdio>
char s[101];
int main(){
    int n,ans=0;
    scanf("%s%d",s,&n);
    for(int i=0;s[i];i++)
        ans=(int)( ((long long)ans*10+s[i]-'0')%n );
    printf("%d\n",ans);
    return 0;
}
/*
只含有 + - *的正整表达式最后取模,等价每步取模:
(a+b)%m = (a%m+b%m)%m;
(a-b)%m = (a%m-b%m + m )%m;
(a*b)%m = (int)((long long)a%m*b%m)%m
input:
123456789987654321 12345
output:
726
