#include<cstdio>//无平方因子数n:不存在k>1,使得x*k^2==n
#include<cmath>
#include<cstring>
int prime[80000],ap,vis[10000001];
int main(){
    int n,m,r;
    scanf("%d%d",&n,&m);
    r=sqrt(m+0.5);
    for(int i=2;i<=r;i++)if(!vis[i]){//筛选取素
        prime[ap++]=i;
        for(int j=i*i;j<=r;j+=i)vis[j]=1;
    }
    memset(vis,0,sizeof(vis));
    for(int i=0,t;prime[i]<=r&&i<ap;i++){//唯一分解,筛选求解
        t=prime[i]*prime[i];
        for(int j=t;j<=m;j+=t)if(j>=n)vis[j-n]=1;
    }
    int cnt=0;
    for(int i=n;i<=m;i++)if(!vis[i-n]){cnt++;}
    printf("%d\n",cnt);
    return 0;
}
/*
1<=n<=m<=10^12 m-n<=10^7
素数定理count(x) 略> x/lnx
input:
1 10000000
output:
6079291
*/
