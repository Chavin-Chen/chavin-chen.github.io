#include<cstdio>
int mod(int a,int n,int m){
    if(!n)return 1;
    int x=mod(a,n/2,m);
    long long ans=(long long)x*x%m;
    if(n%2)ans=ans*a%m;
    return (int)ans;
}
int main(){
    int a,n,m;
    while(scanf("%d%d%d",&a,&n,&m)!=-1){
        printf("%d\n",mod(a,n,m));
    }
    return 0;
}
