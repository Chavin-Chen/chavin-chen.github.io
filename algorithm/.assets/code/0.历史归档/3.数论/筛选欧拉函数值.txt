#include<cstdio>
int a[101];
void phi(int n){
    a[1]=1;
    for(int i=2;i<=n;i++)if(!a[i]){//i一定是素数
        for(int j=i;j<=n;j+=i){    //筛掉i的倍数
            if(!a[j])a[j]=j;
            a[j]=a[j]/i*(i-1);     //计算n(1-1/pi);
        }
    }
}
int main(){
    int n;
    scanf("%d",&n);
    phi(n);
    for(int i=1;i<=n;i++)
        printf(i==n?"%d\n":"%d ",a[i]);
    return 0;
}
/*
1~n的欧拉函数值
